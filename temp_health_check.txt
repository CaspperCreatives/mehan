export const healthCheck: HttpsFunction = onRequest(async (request, response) => { response.json({ status: "healthy", service: "linkedin-extension-api", version: "1.0.0", timestamp: new Date().toISOString(), envCheck: { hasOpenAIKey: !!process.env.OPENAI_API_KEY, openAIKeyLength: process.env.OPENAI_API_KEY ? process.env.OPENAI_API_KEY.length : 0, openAIKeyStart: process.env.OPENAI_API_KEY ? process.env.OPENAI_API_KEY.substring(0, 20) + "..." : "none" } }); });
